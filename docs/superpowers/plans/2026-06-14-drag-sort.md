> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 GridPickerPopup 组件添加自定义类目和支付方式的拖拽排序功能

**Architecture:** 使用自定义触摸事件实现拖拽排序，通过 `useDragSort` 组合式函数封装拖拽逻辑，GridPickerPopup.vue 负责 UI 渲染和调用。预设条目和新增按钮不可拖拽，自定义条目通过磁吸效果限制在有效区域内。

**Tech Stack:** Vue 3 Composition API, uni-app touch events, UnoCSS, wot-design-uni

// src/composables/types/drag-sort.ts

/** 拖拽排序配置项 */
export interface DragSortOptions {
  /** 可排序的项目列表（包含预设和自定义） */
  items: Ref<Array<{ id?: string, isDefault?: boolean }>>
  /** 预设项目数量（不可拖拽） */
  presetCount: Ref<number>
  /** 是否有新增按钮（固定在最后） */
  hasAddButton: Ref<boolean>
  /** 排序完成回调，参数为自定义条目的新 ID 顺序 */
  onSortEnd: (newOrder: string[]) => Promise<void>
  /** 是否禁用拖拽 */
  disabled?: Ref<boolean>
  /** 网格列数，默认 3 */
  columns?: number
}

/** 拖拽状态 */
export interface DragState {
  /** 是否正在拖拽 */
  isDragging: boolean
  /** 被拖拽项的原始索引 */
  dragIndex: number
  /** 目标位置索引 */
  targetIndex: number
  /** 手指当前位置（相对于视口） */
  position: { x: number, y: number }
  /** 触摸点相对于元素左上角的偏移 */
  offset: { x: number, y: number }
  /** 被拖拽元素的尺寸 */
  elementRect: { width: number, height: number }
}

/** 拖拽排序返回值 */
export interface DragSortReturn {
  /** 拖拽状态（只读） */
  dragState: Readonly<Ref<DragState>>
  /** 获取项目的绑定属性 */
  getItemProps: (index: number) => {
    onTouchstart: (e: TouchEvent) => void
    class: string[]
    style: Record<string, string>
  }
  /** 获取容器的绑定属性 */
  containerProps: {
    onTouchmove: (e: TouchEvent) => void
    onTouchend: () => void
    onTouchcancel: () => void
  }
}
```

// src/composables/useDragSort.ts
import type { DragSortOptions, DragSortReturn, DragState } from '@/composables/types/drag-sort'

export function useDragSort(options: DragSortOptions): DragSortReturn {
  const {
    items,
    presetCount,
    hasAddButton,
    onSortEnd,
    disabled = ref(false),
    columns = 3,
  } = options

  // 拖拽状态
  const dragState = ref<DragState>({
    isDragging: false,
    dragIndex: -1,
    targetIndex: -1,
    position: { x: 0, y: 0 },
    offset: { x: 0, y: 0 },
    elementRect: { width: 0, height: 0 },
  })

  // 项目位置信息缓存
  const itemRects = ref<Array<{ top: number, left: number, width: number, height: number }>>([])

  // 防抖定时器
  let sortTimer: ReturnType<typeof setTimeout> | null = null

  return {
    dragState: readonly(dragState),
    getItemProps,
    containerProps,
  }
}
```

在 `useDragSort` 函数内添加：

```typescript
  /** 判断是否可拖拽 */
  function isDraggable(index: number): boolean {
    if (disabled.value) return false
    // 预设条目不可拖拽
    if (index < presetCount.value) return false
    // 新增按钮不可拖拽
    if (hasAddButton.value && index >= items.value.length) return false
    return true
  }

  /** 刷新项目位置信息 */
  function refreshItemRects() {
    const query = uni.createSelectorQuery()
    query.selectAll('.drag-sort-item').boundingClientRect()
    query.exec((res) => {
      if (res?.[0]) {
        itemRects.value = res[0] as Array<{ top: number, left: number, width: number, height: number }>
      }
    })
  }

  /** 获取项目属性绑定 */
  function getItemProps(index: number) {
    return {
      onTouchstart: (e: TouchEvent) => handleTouchStart(e, index),
      class: [
        'drag-sort-item',
        dragState.value.isDragging && dragState.value.dragIndex === index ? 'drag-sort-item--dragging' : '',
        dragState.value.isDragging && dragState.value.targetIndex === index ? 'drag-sort-item--placeholder' : '',
      ].filter(Boolean),
      style: getDragItemStyle(index),
    }
  }

  /** 获取拖拽项的样式 */
  function getDragItemStyle(index: number): Record<string, string> {
    if (!dragState.value.isDragging) return {}
    // 被拖拽的元素：浮动预览
    if (dragState.value.dragIndex === index) {
      return {
        position: 'fixed',
        left: `${dragState.value.position.x - dragState.value.offset.x}px`,
        top: `${dragState.value.position.y - dragState.value.offset.y}px`,
        width: `${dragState.value.elementRect.width}px`,
        height: `${dragState.value.elementRect.height}px`,
        zIndex: '9999',
        opacity: '0.9',
        boxShadow: '0 8px 24px rgba(0, 0, 0, 0.2)',
        transition: 'none',
      }
    }
    // 目标位置：半透明占位符
    if (dragState.value.targetIndex === index) {
      return {
        opacity: '0.4',
        border: '2px dashed #999',
      }
    }
    return {}
  }

  /** 处理触摸开始 */
  function handleTouchStart(e: TouchEvent, index: number) {
    if (!isDraggable(index)) return

    const touch = e.touches[0]
    if (!touch) return

    // 刷新位置信息
    refreshItemRects()

    // 记录初始状态
    const rect = itemRects.value[index]
    if (!rect) return

    dragState.value = {
      isDragging: true,
      dragIndex: index,
      targetIndex: index,
      position: { x: touch.clientX, y: touch.clientY },
      offset: {
        x: touch.clientX - rect.left,
        y: touch.clientY - rect.top,
      },
      elementRect: {
        width: rect.width,
        height: rect.height,
      },
    }
  }
```

- [ ] **Step 3: 实现触摸移动事件处理**

在 `useDragSort` 函数内添加：

```typescript
  /** 获取容器属性绑定 */
  const containerProps = {
    onTouchmove: handleTouchMove,
    onTouchend: handleTouchEnd,
    onTouchcancel: handleTouchCancel,
  }

  /** 处理触摸移动 */
  function handleTouchMove(e: TouchEvent) {
    if (!dragState.value.isDragging) return
    e.preventDefault()

    const touch = e.touches[0]
    if (!touch) return

    // 更新手指位置
    dragState.value.position = {
      x: touch.clientX,
      y: touch.clientY,
    }

    // 计算目标位置
    const newTargetIndex = calculateTargetIndex(touch.clientX, touch.clientY)
    if (newTargetIndex !== -1) {
      dragState.value.targetIndex = applyConstraints(newTargetIndex)
    }
  }

  /** 计算目标位置索引 */
  function calculateTargetIndex(touchX: number, touchY: number): number {
    if (itemRects.value.length === 0) return -1

    for (let i = 0; i < itemRects.value.length; i++) {
      const rect = itemRects.value[i]
      if (
        touchX >= rect.left
        && touchX <= rect.left + rect.width
        && touchY >= rect.top
        && touchY <= rect.top + rect.height
      ) {
        return i
      }
    }

    // 如果没命中任何元素，找最近的
    let minDist = Infinity
    let nearest = -1
    for (let i = 0; i < itemRects.value.length; i++) {
      const rect = itemRects.value[i]
      const centerX = rect.left + rect.width / 2
      const centerY = rect.top + rect.height / 2
      const dist = Math.hypot(touchX - centerX, touchY - centerY)
      if (dist < minDist) {
        minDist = dist
        nearest = i
      }
    }
    return nearest
  }

  /** 应用约束：磁吸效果 */
  function applyConstraints(targetIndex: number): number {
    const minIndex = presetCount.value
    const maxIndex = hasAddButton.value ? items.value.length - 1 : items.value.length

    if (targetIndex < minIndex) return minIndex
    if (targetIndex > maxIndex) return maxIndex
    return targetIndex
  }
```

```typescript
  /** 处理触摸结束 */
  function handleTouchEnd() {
    if (!dragState.value.isDragging) return

    const { dragIndex, targetIndex } = dragState.value

    // 如果位置发生变化，执行排序
    if (dragIndex !== targetIndex) {
      const customItems = items.value.filter(item => !item.isDefault)
      const customIds = customItems.map(item => item.id!)

      // 计算新的排序
      const dragId = items.value[dragIndex]?.id
      if (dragId) {
        // 从原位置移除
        const fromCustomIndex = customIds.indexOf(dragId)
        if (fromCustomIndex !== -1) {
          customIds.splice(fromCustomIndex, 1)
        }

        // 计算在自定义列表中的目标位置
        let toCustomIndex = targetIndex - presetCount.value
        toCustomIndex = Math.max(0, Math.min(toCustomIndex, customIds.length))
        customIds.splice(toCustomIndex, 0, dragId)

        // 防抖调用排序 API
        if (sortTimer) clearTimeout(sortTimer)
        sortTimer = setTimeout(() => {
          onSortEnd(customIds)
        }, 300)
      }
    }

    // 重置拖拽状态
    dragState.value = {
      isDragging: false,
      dragIndex: -1,
      targetIndex: -1,
      position: { x: 0, y: 0 },
      offset: { x: 0, y: 0 },
      elementRect: { width: 0, height: 0 },
    }
  }

  /** 处理触摸取消 */
  function handleTouchCancel() {
    // 取消时直接重置状态
    dragState.value = {
      isDragging: false,
      dragIndex: -1,
      targetIndex: -1,
      position: { x: 0, y: 0 },
      offset: { x: 0, y: 0 },
      elementRect: { width: 0, height: 0 },
    }
  }
```
